3) Make a dedicated handler for Sync Plan (inside component)
const handleSyncPlan = async () => {
  if (busy) return;
  setSyncNote('Syncing…');
  setBusy(true);
  try {
    await getEntitlements();       // uses your existing API helper
    const ts = new Date();
    setLastSyncAt(ts);
    setSyncNote('Synced just now');
    setTimeout(() => setSyncNote(''), 2500); // fade note after a moment
    console.log('[SyncPlan] refreshed at', ts.toISOString());
  } catch (e) {
    console.warn('[SyncPlan] failed', e);
    Alert.alert('Sync failed', 'Could not refresh your plan. Please try again.');
    setSyncNote('');
  } finally {
    setBusy(false);
  }
};
